from flask import Flask, request
import face_recognition
import cv2
import os
import openpyxl
from datetime import datetime
import numpy as np

# ========== CONFIG ==========
USER_FILE = "authorized_users.csv"
EXCEL_FILE = "Book3.xlsx"
KNOWN_FACES_DIR = "known_faces"

# ========== GLOBALS ==========
known_face_encodings = []
known_face_names = []
card_to_name = {}

app = Flask(__name__)

# ---------- Load data ----------
def load_faces():
    global known_face_encodings, known_face_names
    known_face_encodings.clear()
    known_face_names.clear()
    for filename in os.listdir(KNOWN_FACES_DIR):
        if filename.endswith(".jpg"):
            path = os.path.join(KNOWN_FACES_DIR, filename)
            img = face_recognition.load_image_file(path)
            enc = face_recognition.face_encodings(img)
            if enc:
                known_face_encodings.append(enc[0])
                known_face_names.append(filename.split(".")[0])
            else:
                print(f"[WARN] No face in {filename}")

def load_user_records():
    global card_to_name
    card_to_name = {}
    if os.path.exists(USER_FILE):
        with open(USER_FILE, "r") as f:
            for line in f:
                parts = line.strip().split(",")
                if len(parts) == 2:
                    card_id, name = parts
                    card_to_name[card_id] = name

def save_to_excel(name, card_id, status):
    now = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    try:
        wb = openpyxl.load_workbook(EXCEL_FILE)
    except FileNotFoundError:
        wb = openpyxl.Workbook()
    sheet = wb.active
    if sheet.max_row == 1:
        sheet.append(["Timestamp", "Card ID", "Name", "Status"])
    sheet.append([now, card_id, name, status])
    wb.save(EXCEL_FILE)

def rotate_image(image, angle):
    (h, w) = image.shape[:2]
    center = (w // 2, h // 2)
    matrix = cv2.getRotationMatrix2D(center, angle, 1.0)
    return cv2.warpAffine(image, matrix, (w, h))

# ---------- Main API ----------
@app.route("/api/rfid")
def rfid_api():
    card_id = request.args.get("uid", "").strip()
    if not card_id:
        return "BAD_REQUEST", 400
    print(f"[RFID] Card detected: {card_id}")

    if card_id not in card_to_name:
        print("‚ùå Unauthorized card")
        save_to_excel("Unknown", card_id, "Unauthorized")
        return "UNAUTH"

    expected_name = card_to_name[card_id]
    print(f"[INFO] Expected: {expected_name}")
    cap = cv2.VideoCapture(0)
    matched = False
    matched_name = "Unknown"

    for _ in range(3):
        ret, frame = cap.read()
        if not ret:
            continue
        rgb_frame = frame[:, :, ::-1]

        for angle in [0, -15, 15, -30, 30]:
            rotated = rotate_image(rgb_frame, angle)
            face_locations = face_recognition.face_locations(rotated)
            face_encodings = face_recognition.face_encodings(rotated, face_locations)

            for face_encoding in face_encodings:
                distances = face_recognition.face_distance(known_face_encodings, face_encoding)
                if len(distances) == 0:
                    continue
                best_idx = np.argmin(distances)
                if distances[best_idx] < 0.55:
                    matched_name = known_face_names[best_idx]
                    if matched_name == expected_name:
                        matched = True
                        break
            if matched:
                break
        if matched:
            break

    cap.release()
    cv2.destroyAllWindows()

    if matched:
        print(f"‚úÖ Match: {matched_name}")
        save_to_excel(matched_name, card_id, "Authorized")
        return "FACE_OK"
    else:
        print("‚ùå Face not matched")
        save_to_excel("Unknown", card_id, "Face Not Match")
        return "FACE_FAIL"

@app.route("/api/rfid")
def rfid_api():
    card_id = request.args.get("uid", "").strip()
    if not card_id:
        return "BAD_REQUEST", 400

    append_log(f"[RFID] Card Detected: {card_id}")
    evt = threading.Event()
    pending_results[card_id] = {"event": evt, "result": None}
    request_queue.put(card_id)

    # wait up to 10 seconds for main thread processing
    processed = evt.wait(timeout=10)
    if processed:
        res = pending_results[card_id]["result"]
        del pending_results[card_id]
        return res
    else:
        del pending_results[card_id]
        append_log(f"[ERROR] Processing timeout for {card_id}")
        return "TIMEOUT", 504

# ========== QUEUE POLLING ==========
def poll_queue():
    try:
        card_id = request_queue.get_nowait()
    except queue.Empty:
        root.after(200, poll_queue)
        return

    append_log(f"[QUEUE] Processing card: {card_id}")
    result = handle_card(card_id)

    entry = pending_results.get(card_id)
    if entry:
        entry["result"] = result
        entry["event"].set()

    root.after(200, poll_queue)
